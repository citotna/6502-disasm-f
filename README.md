# 6502-disasm-f
6502 disassembler in C, just 32 lines of code

## Usage
```C
int main() {
	uint8_t code[] = {
	    0xA5,0x1F,0xC9,0x01,0xF0,0x01,0x60,0xA9,0x00,0x85,0x0C,
	    0xA9,0x78,0x85,0x0D,0xA9,0x00,0x85,0x6D,0xA0,0x00,0xB1,
	    0x0C,0x85,0x6E,0x29,0x1F,0xC9,0x05,0xD0,0x40,0xA5,0x6E,
	    0x29,0xE0,0x85,0x6E,0xA6,0x6D,0xE8,0xE4,0x0F,0x90,0x08,
	    0xA5,0x20,0xC9,0x05,0xD0,0x27,0xF0,0x2B,0xA5,0x0C,0x18,
	    0x65,0x0E,0x85,0x72,0xA5,0x0D,0x69,0x00,0x85,0x73,0xB1,
	    0x72,0x85,0x6F,0x29,0xE0,0xC5,0x6E,0xF0,0x02,0x10,0x12,
	    0xA5,0x6F,0x29,0x1F,0xC9,0x05,0xF0,0x0A,0xC9,0x07,0xF0,
	    0x06,0xA9,0x07,0x05,0x6E,0x91,0x0C,0xC8,0xC4,0x0E,0x90,
	    0xB1,0xA5,0x0C,0x18,0x65,0x0E,0x85,0x0C,0x90,0x02,0xE6,
	    0x0D,0xE6,0x6D,0xA5,0x6D,0xC5,0x0F,0x90,0x9C,0x60
  };
	disasm(code,sizeof(code),0xabc9);
  return 1;
}
```
